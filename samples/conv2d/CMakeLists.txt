# buffer property available in ComputeCpp 1.0.3 and onwards.

  snn_executable(
    WITH_SYCL
    TARGET
      on_chip_memory
    SOURCES
      on_chip_memory.cc
    PRIVATE_LIBRARIES
      sycl_dnn_static
      Eigen::Eigen
  )
install(
  TARGETS on_chip_memory
  RUNTIME DESTINATION ${runtime_dest}
)
endif()
